\begin{frame}{Kiến trúc ST-GCN Cải tiến}
    \begin{itemize}
        \item \textbf{Chiều sâu:} Nâng cấp lên 6 ST-GCN Blocks.
        \item \textbf{Adaptive Graph:} Ma trận kề động có thể học được.
        \item \textbf{Quy trình:}
        \begin{enumerate}
            \item Spatial GCN: Học hình dáng tay.
            \item Temporal CNN: Học chuyển động (Kernel=9).
        \end{enumerate}
        \item \textbf{Kỹ thuật:} Residual connection, Batch Norm, Dropout.
    \end{itemize}
\end{frame}
